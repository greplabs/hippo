92    /// Get the best available model from a category
93    pub async fn get_best_available(client: &OllamaClient, category: &[&str]) -> Option<String> {
94        if let Ok(models) = client.list_models().await {
95            let model_names: Vec<String> = models.iter().map(|m| m.name.clone()).collect();
96            for preferred in category {
97                // Check exact match or prefix match (for versioned models)
98                if model_names
99                    .iter()
100                    .any(|m| m == *preferred || m.starts_with(&format!("{}:", preferred)))
101                {
102                    return Some(preferred.to_string());
103                }
104            }
105        }
106        None

330    /// Create a new Ollama client with custom configuration
331    pub fn with_config(config: OllamaConfig) -> Self {
332        // Optimized HTTP client with aggressive connection pooling
333        let client = Client::builder()
334            .timeout(Duration::from_secs(config.timeout_secs))
335            .pool_max_idle_per_host(20) // More idle connections for high throughput
336            .pool_idle_timeout(Duration::from_secs(120)) // Longer idle timeout
337            .tcp_keepalive(Duration::from_secs(30)) // More frequent keepalive
338            .tcp_nodelay(true) // Disable Nagle for lower latency
339            .build()
340            .unwrap_or_else(|_| Client::new());
341
342        Self {
343            client,
344            base_url: config.base_url,
345            embedding_model: config.embedding_model,
346            generation_model: config.generation_model,
347            model_cache: Arc::new(RwLock::new(None)),
348            embedding_cache: Arc::new(RwLock::new(LruCache::new(
349                NonZeroUsize::new(EMBEDDING_CACHE_SIZE).unwrap(),
350            ))),
351            availability_cache: Arc::new(RwLock::new(None)),
352        }
353    }

355    /// Check if Ollama is running and accessible (cached for 30 seconds)
356    pub async fn is_available(&self) -> bool {
357        // Check cache first (30-second TTL for availability)
358        {
359            let cache = self.availability_cache.read();
360            if let Some((cached_at, available)) = *cache {
361                if cached_at.elapsed().as_secs() < 30 {
362                    return available;
363                }
364            }
365        }
366
367        // Cache miss - check availability
368        let available = match self.client.get(&self.base_url).send().await {
369            Ok(resp) => resp.status().is_success(),
370            Err(_) => false,
371        };
372
373        // Update cache
374        {
375            let mut cache = self.availability_cache.write();
376            *cache = Some((Instant::now(), available));
377        }
378
379        available
380    }

401    /// List available models (cached for 5 minutes)
402    pub async fn list_models(&self) -> Result<Vec<OllamaModel>> {
403        // Check cache first
404        {
405            let cache = self.model_cache.read();
406            if let Some(cached) = cache.as_ref() {
407                if cached.cached_at.elapsed().as_secs() < MODEL_CACHE_TTL_SECS {
408                    debug!(
409                        "Returning cached model list ({} models)",
410                        cached.models.len()
411                    );
412                    return Ok(cached.models.clone());
413                }
414            }
415        }
416
417        // Cache miss - fetch from API
418        let url = format!("{}/api/tags", self.base_url);
419        let resp = self
420            .client
421            .get(&url)
422            .send()
423            .await
424            .map_err(|e| HippoError::Other(format!("Failed to list models: {}", e)))?;
425
426        if !resp.status().is_success() {
427            return Err(HippoError::Other("Failed to list models".to_string()));
428        }
429
430        let models: ModelsResponse = resp
431            .json()
432            .await
433            .map_err(|e| HippoError::Other(format!("Failed to parse models: {}", e)))?;
434
435        // Update cache
436        {
437            let mut cache = self.model_cache.write();
438            *cache = Some(ModelCache {
439                models: models.models.clone(),
440                cached_at: Instant::now(),
441            });
442        }
443
444        debug!("Cached {} models from Ollama", models.models.len());
445        Ok(models.models)
446    }

490        }

508

526    /// Generate embedding for a single text (with caching)
527    pub async fn embed_single(&self, text: &str) -> Result<Vec<f32>> {
528        // Create cache key from text hash
529        let cache_key = format!("{}:{:x}", &self.embedding_model, Self::hash_text(text));
530
531        // Check cache first
532        {
533            let cache = self.embedding_cache.read();
534            if let Some(entry) = cache.peek(&cache_key) {
535                if entry.cached_at.elapsed().as_secs() < EMBEDDING_CACHE_TTL_SECS {
536                    debug!("Embedding cache hit for text hash");
537                    return Ok(entry.embedding.clone());
538                }
539            }
540        }
541
542        // Cache miss - generate embedding
543        let embeddings = self.embed(&[text.to_string()]).await?;
544        let embedding = embeddings
545            .into_iter()
546            .next()
547            .ok_or_else(|| HippoError::Other("No embedding returned".to_string()))?;
548
549        // Update cache
550        {
551            let mut cache = self.embedding_cache.write();
552            cache.put(
553                cache_key,
554                EmbeddingCacheEntry {
555                    embedding: embedding.clone(),
556                    cached_at: Instant::now(),
557                },
558            );
559        }
560
561        Ok(embedding)
562    }

564    /// Simple hash function for cache keys
565    fn hash_text(text: &str) -> u64 {
566        use std::collections::hash_map::DefaultHasher;
567        use std::hash::{Hash, Hasher};
568        let mut hasher = DefaultHasher::new();
569        text.hash(&mut hasher);
570        hasher.finish()
571    }

653    /// Stream chat responses (yields chunks as they arrive)
654    pub async fn stream_chat<F>(
655        &self,
656        messages: &[ChatMessage],
657        cancellation_token: CancellationToken,
658        mut on_chunk: F,
659    ) -> Result<String>
660    where
661        F: FnMut(String) + Send,
662    {
663        let url = format!("{}/api/chat", self.base_url);
664
665        let request = ChatRequest {
666            model: self.generation_model.clone(),
667            messages: messages.to_vec(),
668            stream: true, // Enable streaming
669            options: GenerateOptions {
670                temperature: Some(0.7),
671                num_predict: Some(2048),
672                top_p: Some(0.9),
673            },
674        };
675
676        debug!(
677            "Starting streaming chat with model: {}",
678            self.generation_model
679        );
680
681        let resp = self
682            .client
683            .post(&url)
684            .json(&request)
685            .timeout(Duration::from_secs(600)) // 10 minute timeout for streaming (large models)
686            .send()
687            .await
688            .map_err(|e| HippoError::Other(format!("Chat request failed: {}", e)))?;
689
690        if !resp.status().is_success() {
691            let error = resp.text().await.unwrap_or_default();
692            return Err(HippoError::Other(format!("Chat failed: {}", error)));
693        }
694
695        let mut stream = resp.bytes_stream();
696        let mut full_response = String::new();
697        let mut buffer = String::new();
698
699        while let Some(chunk_result) = stream.next().await {
700            // Check for cancellation
701            if cancellation_token.is_cancelled() {
702                debug!("Stream cancelled by user");
703                return Err(HippoError::Other("Stream cancelled".to_string()));
704            }
705
706            let chunk =
707                chunk_result.map_err(|e| HippoError::Other(format!("Stream error: {}", e)))?;
708
709            buffer.push_str(&String::from_utf8_lossy(&chunk));
710
711            // Process complete JSON lines
712            while let Some(newline_pos) = buffer.find('\n') {
713                let line = buffer[..newline_pos].trim().to_string();
714                buffer = buffer[newline_pos + 1..].to_string();
715
716                if line.is_empty() {
717                    continue;
718                }
719
720                // Parse the streaming response
721                match serde_json::from_str::<ChatResponse>(&line) {
722                    Ok(chat_resp) => {
723                        let content = &chat_resp.message.content;
724                        if !content.is_empty() {
725                            full_response.push_str(content);
726                            on_chunk(content.clone());
727                        }
728
729                        if chat_resp.done {
730                            debug!("Stream completed successfully");
731                            return Ok(full_response);
732                        }
733                    }
734                    Err(e) => {
735                        warn!("Failed to parse streaming chunk: {} - Line: {}", e, line);
736                    }
737                }
738            }
739        }
740
741        Ok(full_response)
742    }

744    /// Stream text generation (yields chunks as they arrive)
745    pub async fn stream_generate<F>(
746        &self,
747        prompt: &str,
748        system: Option<&str>,
749        cancellation_token: CancellationToken,
750        mut on_chunk: F,
751    ) -> Result<String>
752    where
753        F: FnMut(String) + Send,
754    {
755        let url = format!("{}/api/generate", self.base_url);
756
757        let request = GenerateRequest {
758            model: self.generation_model.clone(),
759            prompt: prompt.to_string(),
760            stream: true, // Enable streaming
761            system: system.map(String::from),
762            context: None,
763            options: GenerateOptions {
764                temperature: Some(0.7),
765                num_predict: Some(1024),
766                top_p: Some(0.9),
767            },
768        };
769
770        debug!(
771            "Starting streaming generation with model: {}",
772            self.generation_model
773        );
774
775        let resp = self
776            .client
777            .post(&url)
778            .json(&request)
779            .timeout(Duration::from_secs(600)) // 10 minute timeout for streaming (large models)
780            .send()
781            .await
782            .map_err(|e| HippoError::Other(format!("Generate request failed: {}", e)))?;
783
784        if !resp.status().is_success() {
785            let status = resp.status();
786            let error = resp.text().await.unwrap_or_default();
787            return Err(HippoError::Other(format!(
788                "Generation failed ({}): {}. Make sure '{}' model is installed.",
789                status, error, self.generation_model
790            )));
791        }
792
793        let mut stream = resp.bytes_stream();
794        let mut full_response = String::new();
795        let mut buffer = String::new();
796
797        while let Some(chunk_result) = stream.next().await {
798            // Check for cancellation
799            if cancellation_token.is_cancelled() {
800                debug!("Stream cancelled by user");
801                return Err(HippoError::Other("Stream cancelled".to_string()));
802            }
803
804            let chunk =
805                chunk_result.map_err(|e| HippoError::Other(format!("Stream error: {}", e)))?;
806
807            buffer.push_str(&String::from_utf8_lossy(&chunk));
808
809            // Process complete JSON lines
810            while let Some(newline_pos) = buffer.find('\n') {
811                let line = buffer[..newline_pos].trim().to_string();
812                buffer = buffer[newline_pos + 1..].to_string();
813
814                if line.is_empty() {
815                    continue;
816                }
817
818                // Parse the streaming response
819                match serde_json::from_str::<GenerateResponse>(&line) {
820                    Ok(gen_resp) => {
821                        let content = &gen_resp.response;
822                        if !content.is_empty() {
823                            full_response.push_str(content);
824                            on_chunk(content.clone());
825                        }
826
827                        if gen_resp.done {
828                            debug!("Stream completed successfully");
829                            return Ok(full_response);
830                        }
831                    }
832                    Err(e) => {
833                        warn!("Failed to parse streaming chunk: {} - Line: {}", e, line);
834                    }
835                }
836            }
837        }
838
839        Ok(full_response)
840    }

937            "Extract entities from:\n\n{}",
938            &text[..text.len().min(4000)]

940        let response = self.generate(&prompt, Some(system_prompt)).await?;
941
942        // Try to parse as JSON array
943        if let Ok(entities) = serde_json::from_str::<Vec<String>>(&response) {
944            Ok(entities)
945        } else {
946            // Fallback: split by newlines/commas
947            Ok(response
948                .lines()
949                .flat_map(|line| line.split(','))
950                .map(|s| {
951                    s.trim()
952                        .trim_matches(|c| c == '"' || c == '[' || c == ']')
953                        .to_string()
954                })
955                .filter(|s| !s.is_empty() && s.len() > 1)
956                .collect())
957        }

977    /// Parse analysis response from JSON
978    fn parse_analysis_response(&self, response: &str) -> Result<LocalAnalysis> {
979        // Try to extract JSON from the response
980        let json_str = if let Some(start) = response.find('{') {
981            if let Some(end) = response.rfind('}') {
982                &response[start..=end]
983            } else {
984                response
985            }
986        } else {
987            response
988        };
989
990        match serde_json::from_str::<serde_json::Value>(json_str) {
991            Ok(json) => Ok(LocalAnalysis {
992                // Truncate summary to save storage space
993                summary: json
994                    .get("summary")
995                    .and_then(|v| v.as_str())
996                    .unwrap_or("No summary available")
997                    .chars()
998                    .take(MAX_AI_SUMMARY_CHARS)
999                    .collect(),
1000                key_topics: json
1001                    .get("key_topics")
1002                    .and_then(|v| v.as_array())
1003                    .map(|arr| {
1004                        arr.iter()
1005                            .filter_map(|v| v.as_str().map(String::from))
1006                            .collect()
1007                    })
1008                    .unwrap_or_default(),
1009                suggested_tags: json
1010                    .get("suggested_tags")
1011                    .and_then(|v| v.as_array())
1012                    .map(|arr| {
1013                        arr.iter()
1014                            .filter_map(|v| v.as_str().map(String::from))
1015                            .collect()
1016                    })
1017                    .unwrap_or_default(),
1018                document_type: json
1019                    .get("document_type")
1020                    .and_then(|v| v.as_str())
1021                    .map(String::from),
1022                language: json
1023                    .get("language")
1024                    .and_then(|v| v.as_str())
1025                    .map(String::from),
1026            }),
1027            Err(e) => {
1028                warn!("Failed to parse analysis JSON: {}", e);
1029                // Return a basic analysis from the raw response (truncated for storage)
1030                Ok(LocalAnalysis {
1031                    summary: response.chars().take(MAX_AI_SUMMARY_CHARS).collect(),
1032                    key_topics: vec![],
1033                    suggested_tags: vec![],
1034                    document_type: None,
1035                    language: None,
1036                })
1037            }
1038        }
1039    }

1061    /// Caption an image using a vision model (llava:7b)
1062    pub async fn caption_image(&self, image_path: &std::path::Path) -> Result<String> {
1063        // Read image file and convert to base64
1064        let image_data = std::fs::read(image_path)
1065            .map_err(|e| HippoError::Other(format!("Failed to read image: {}", e)))?;
1066        let base64_image = base64::engine::general_purpose::STANDARD.encode(&image_data);
1067
1068        // Use llava:7b model for vision
1069        let vision_model = "llava:7b";
1070
1071        // Check if vision model is available
1072        if !self.has_model(vision_model).await {
1073            return Err(HippoError::Other(format!(
1074                "Vision model '{}' not installed. Run: ollama pull {}",
1075                vision_model, vision_model
1076            )));
1077        }
1078
1079        // Create chat request with image
1080        let url = format!("{}/api/chat", self.base_url);
1081
1082        let messages = vec![ChatMessage::with_images(
1083            "user",
1084            "Describe this image in detail. What do you see?",
1085            vec![base64_image],
1086        )];
1087
1088        let request = ChatRequest {
1089            model: vision_model.to_string(),
1090            messages,
1091            stream: false,
1092            options: GenerateOptions {
1093                temperature: Some(0.7),
1094                num_predict: Some(512),
1095                top_p: Some(0.9),
1096            },
1097        };
1098
1099        debug!("Generating image caption with model: {}", vision_model);
1100
1101        let resp = self
1102            .client
1103            .post(&url)
1104            .json(&request)
1105            .send()
1106            .await
1107            .map_err(|e| HippoError::Other(format!("Vision request failed: {}", e)))?;
1108
1109        if !resp.status().is_success() {
1110            let error = resp.text().await.unwrap_or_default();
1111            return Err(HippoError::Other(format!("Vision failed: {}", error)));
1112        }
1113
1114        let chat_resp: ChatResponse = resp
1115            .json()
1116            .await
1117            .map_err(|e| HippoError::Other(format!("Failed to parse vision response: {}", e)))?;
1118
1119        // Truncate caption for storage optimization
1120        let caption: String = chat_resp
1121            .message
1122            .content
1123            .chars()
1124            .take(MAX_AI_CAPTION_CHARS)
1125            .collect();
1126        Ok(caption)
1127    }
