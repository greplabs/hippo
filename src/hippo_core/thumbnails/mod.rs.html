1//! Thumbnail generation and caching for fast image, video, PDF, and document previews
2//!
3//! Features:
4//! - Disk caching for persistence across restarts
5//! - In-memory LRU cache for hot thumbnails
6//! - Automatic cache eviction when memory limit reached
7//! - Support for images, videos (via ffmpeg), PDFs (via pdfium), and Office documents
8

20/// Default thumbnail size (width and height)
21pub const THUMBNAIL_SIZE: u32 = 256;
22
23/// JPEG quality for thumbnails (60% provides good balance of size/quality)
24/// Default ~75% creates 8-12KB thumbnails, 60% creates 4-6KB (~50% smaller)
25pub const THUMBNAIL_JPEG_QUALITY: u8 = 60;
26
27/// Default LRU cache capacity (number of thumbnails to keep in memory)
28/// Increased for better hit rate on large photo libraries
29pub const DEFAULT_CACHE_CAPACITY: usize = 2000;
30
31/// Maximum memory usage for in-memory cache (100MB - modern systems can handle this)
32/// Each thumbnail is ~5KB, so 100MB = ~20,000 thumbnails
33pub const MAX_CACHE_MEMORY_BYTES: usize = 100 * 1024 * 1024;

35/// Thumbnail manager for generating and caching image thumbnails
36pub struct ThumbnailManager {
37    cache_dir: PathBuf,
38    size: u32,
39    /// In-memory LRU cache for recently accessed thumbnails

42    cache_memory_usage: Mutex<usize>,
43}

61    /// Create a new thumbnail manager with a custom cache directory
62    pub fn with_cache_dir(cache_dir: PathBuf) -> Result<Self> {
63        // Create cache directory if it doesn't exist
64        if !cache_dir.exists() {
65            fs::create_dir_all(&cache_dir)?;
66        }
67
68        Ok(Self {
69            cache_dir,
70            size: THUMBNAIL_SIZE,
71            memory_cache: Mutex::new(LruCache::new(

73            )),
74            cache_memory_usage: Mutex::new(0),
75        })
76    }

78    /// Get thumbnail data from memory cache or disk
79    pub fn get_thumbnail_data(&self, image_path: &Path) -> Result<Vec<u8>> {
80        let cache_key = self.get_thumbnail_name(image_path);
81
82        // Check memory cache first
83        {
84            let mut cache = self.memory_cache.lock();
85            if let Some(cached) = cache.get(&cache_key) {
86                debug!("Thumbnail cache hit (memory): {}", cache_key);
87                return Ok(cached.data.clone());
88            }
89        }
90
91        // Not in memory, check disk cache
92        let thumbnail_path = self.cache_dir.join(&cache_key);
93        if thumbnail_path.exists() {
94            let data = fs::read(&thumbnail_path)?;
95            self.add_to_memory_cache(cache_key, data.clone());
96            debug!("Thumbnail cache hit (disk): {:?}", thumbnail_path);
97            return Ok(data);
98        }
99
100        // Need to generate
101        let generated_path = self.generate_thumbnail(image_path)?;
102        let data = fs::read(&generated_path)?;
103        self.add_to_memory_cache(self.get_thumbnail_name(image_path), data.clone());
104        Ok(data)
105    }

107    /// Add a thumbnail to the memory cache with memory limit enforcement
108    fn add_to_memory_cache(&self, key: String, data: Vec<u8>) {
109        let data_size = data.len();
110
111        // Check if adding this would exceed memory limit
112        let mut memory_usage = self.cache_memory_usage.lock();
113        let mut cache = self.memory_cache.lock();
114
115        // Evict entries if necessary to stay under memory limit
116        while *memory_usage + data_size > MAX_CACHE_MEMORY_BYTES && !cache.is_empty() {
117            if let Some((_, evicted)) = cache.pop_lru() {
118                *memory_usage = memory_usage.saturating_sub(evicted.data.len());
119            }
120        }
121
122        // Add to cache
123        if let Some(old) = cache.put(key, CachedThumbnail { data }) {
124            *memory_usage = memory_usage.saturating_sub(old.data.len());
125        }
126        *memory_usage += data_size;
127    }

132        let mut memory_usage = self.cache_memory_usage.lock();
133        cache.clear();
134        *memory_usage = 0;
135        debug!("Memory cache cleared");
136    }

161    /// Generate a thumbnail for an image file
162    /// Returns the path to the thumbnail if successful
163    pub fn generate_thumbnail(&self, image_path: &Path) -> Result<PathBuf> {
164        // Generate a unique filename based on the original path
165        let thumbnail_name = self.get_thumbnail_name(image_path);
166        let thumbnail_path = self.cache_dir.join(&thumbnail_name);
167
168        // Check if thumbnail already exists and is newer than source
169        if thumbnail_path.exists() {
170            if let (Ok(src_meta), Ok(thumb_meta)) =
171                (fs::metadata(image_path), fs::metadata(&thumbnail_path))
172            {
173                if let (Ok(src_time), Ok(thumb_time)) = (src_meta.modified(), thumb_meta.modified())
174                {
175                    if thumb_time >= src_time {
176                        debug!("Using cached thumbnail: {:?}", thumbnail_path);
177                        return Ok(thumbnail_path);
178                    }
179                }
180            }
181        }
182
183        // Load and resize the image
184        debug!("Generating thumbnail for: {:?}", image_path);
185        let img = image::open(image_path).map_err(|e| {
186            HippoError::Other(format!("Failed to open image {:?}: {}", image_path, e))
187        })?;
188
189        // Create thumbnail (maintains aspect ratio)
190        let thumbnail = self.create_thumbnail(&img);
191
192        // Save thumbnail as JPEG with optimized quality (60% = ~50% smaller files)
193        let mut output = fs::File::create(&thumbnail_path)?;
194        let rgb = thumbnail.to_rgb8();
195        let mut encoder =
196            image::codecs::jpeg::JpegEncoder::new_with_quality(&mut output, THUMBNAIL_JPEG_QUALITY);
197        encoder
198            .encode(
199                &rgb,
200                rgb.width(),
201                rgb.height(),
202                image::ExtendedColorType::Rgb8,
203            )
204            .map_err(|e| HippoError::Other(format!("Failed to save thumbnail: {}", e)))?;
205
206        debug!("Thumbnail saved: {:?}", thumbnail_path);
207        Ok(thumbnail_path)
208    }

231    pub fn clear_cache(&self) -> Result<()> {
232        if self.cache_dir.exists() {
233            for entry in fs::read_dir(&self.cache_dir)?.flatten() {
234                let path = entry.path();
235                if path.is_file() {
236                    if let Err(e) = fs::remove_file(&path) {
237                        warn!("Failed to delete thumbnail {:?}: {}", path, e);
238                    }
239                }
240            }
241        }
242        Ok(())
243    }

261        Ok(ThumbnailStats { count, total_size })

264    /// Generate a unique thumbnail filename from the original path
265    fn get_thumbnail_name(&self, image_path: &Path) -> String {
266        use sha2::{Digest, Sha256};
267
268        // Hash the full path to create a unique filename
269        let path_str = image_path.to_string_lossy();
270        let mut hasher = Sha256::new();
271        hasher.update(path_str.as_bytes());
272        let hash = hasher.finalize();
273        let hash_hex = hex::encode(&hash[..16]); // Use first 16 bytes (32 hex chars)
274
275        format!("{}.jpg", hash_hex)
276    }

284    /// Generate a thumbnail for a video file by extracting a frame
285    /// Returns the path to the thumbnail if successful
286    pub fn generate_video_thumbnail(&self, video_path: &Path) -> Result<PathBuf> {
287        // Check if ffmpeg is available
288        if !is_ffmpeg_available() {
289            return Err(HippoError::Other(
290                "ffmpeg not available - cannot generate video thumbnails".to_string(),
291            ));
292        }
293
294        // Generate a unique filename based on the original path
295        let thumbnail_name = self.get_thumbnail_name(video_path);
296        let thumbnail_path = self.cache_dir.join(&thumbnail_name);
297
298        // Check if thumbnail already exists and is newer than source
299        if thumbnail_path.exists() {
300            if let (Ok(src_meta), Ok(thumb_meta)) =
301                (fs::metadata(video_path), fs::metadata(&thumbnail_path))
302            {
303                if let (Ok(src_time), Ok(thumb_time)) = (src_meta.modified(), thumb_meta.modified())
304                {
305                    if thumb_time >= src_time {
306                        debug!("Using cached video thumbnail: {:?}", thumbnail_path);
307                        return Ok(thumbnail_path);
308                    }
309                }
310            }
311        }
312
313        debug!("Generating video thumbnail for: {:?}", video_path);
314
315        // Try to extract frame at 2 seconds, fall back to 0 if video is short
316        let seek_times = ["2", "1", "0"];
317
318        for seek_time in seek_times {
319            let output = Command::new("ffmpeg")
320                .args([
321                    "-y", // Overwrite output
322                    "-ss", seek_time, // Seek to timestamp
323                    "-i",
324                ])
325                .arg(video_path)
326                .args([
327                    "-vframes",
328                    "1", // Extract single frame
329                    "-vf",
330                    &format!(
331                        "scale={}:{}:force_original_aspect_ratio=decrease",
332                        self.size, self.size
333                    ),
334                    "-f",
335                    "image2", // Output format
336                ])
337                .arg(&thumbnail_path)
338                .output();
339
340            match output {
341                Ok(result) if result.status.success() && thumbnail_path.exists() => {
342                    debug!("Video thumbnail saved: {:?}", thumbnail_path);
343                    return Ok(thumbnail_path);
344                }
345                Ok(result) => {
346                    debug!(
347                        "ffmpeg failed at seek={}: {}",
348                        seek_time,
349                        String::from_utf8_lossy(&result.stderr)
350                    );
351                    // Try next seek time
352                }
353                Err(e) => {
354                    warn!("Failed to run ffmpeg: {}", e);
355                    break;
356                }
357            }
358        }
359
360        Err(HippoError::Other(format!(
361            "Failed to generate video thumbnail for {:?}",
362            video_path
363        )))
364    }

407    /// Render PDF first page using pdfium-render
408    fn render_pdf_with_pdfium(&self, pdf_path: &Path, output_path: &Path) -> Result<()> {
409        use pdfium_render::prelude::*;
410
411        // Initialize pdfium
412        let pdfium = Pdfium::new(
413            Pdfium::bind_to_library(Pdfium::pdfium_platform_library_name_at_path("./"))
414                .or_else(|_| Pdfium::bind_to_system_library())
415                .map_err(|e| HippoError::Other(format!("Failed to load pdfium library: {}", e)))?,
416        );
417
418        // Load the PDF document
419        let document = pdfium
420            .load_pdf_from_file(pdf_path, None)
421            .map_err(|e| HippoError::Other(format!("Failed to load PDF: {}", e)))?;
422
423        // Get the first page
424        let page = document
425            .pages()
426            .get(0)
427            .map_err(|e| HippoError::Other(format!("Failed to get first page: {}", e)))?;
428
429        // Calculate render dimensions maintaining aspect ratio
430        let page_width = page.width().value;
431        let page_height = page.height().value;
432        let aspect_ratio = page_width / page_height;
433
434        let (render_width, render_height) = if aspect_ratio > 1.0 {
435            // Landscape
436            (self.size, (self.size as f32 / aspect_ratio) as u32)
437        } else {
438            // Portrait or square
439            ((self.size as f32 * aspect_ratio) as u32, self.size)
440        };
441
442        // Render the page to a bitmap
443        let bitmap = page
444            .render_with_config(
445                &PdfRenderConfig::new()
446                    .set_target_width(render_width as i32)
447                    .set_target_height(render_height as i32)
448                    .rotate_if_landscape(PdfPageRenderRotation::None, true),
449            )
450            .map_err(|e| HippoError::Other(format!("Failed to render page: {}", e)))?;
451
452        // Convert to image and save with optimized quality
453        let dynamic_image = bitmap.as_image();
454        let thumbnail = self.create_thumbnail(&dynamic_image);
455
456        let mut output = fs::File::create(output_path)?;
457        let rgb = thumbnail.to_rgb8();
458        let mut encoder =
459            image::codecs::jpeg::JpegEncoder::new_with_quality(&mut output, THUMBNAIL_JPEG_QUALITY);
460        encoder
461            .encode(
462                &rgb,
463                rgb.width(),
464                rgb.height(),
465                image::ExtendedColorType::Rgb8,
466            )
467            .map_err(|e| HippoError::Other(format!("Failed to save PDF thumbnail: {}", e)))?;
468
469        Ok(())
470    }

517    /// Extract thumbnail from Office Open XML document (ZIP archive)
518    fn extract_office_thumbnail_from_zip(&self, doc_path: &Path, output_path: &Path) -> Result<()> {
519        let file = fs::File::open(doc_path)?;
520        let mut archive = zip::ZipArchive::new(file)
521            .map_err(|e| HippoError::Other(format!("Failed to open ZIP archive: {}", e)))?;
522
523        // Common thumbnail locations in Office documents
524        let thumbnail_paths = [
525            "docProps/thumbnail.jpeg",
526            "docProps/thumbnail.jpg",
527            "docProps/thumbnail.png",
528            "docProps/thumbnail.emf",
529            "docProps/thumbnail.wmf",
530        ];
531
532        // Try each possible thumbnail path
533        for thumb_path in &thumbnail_paths {
534            if let Ok(mut file) = archive.by_name(thumb_path) {
535                debug!("Found embedded thumbnail at: {}", thumb_path);
536
537                // Read the thumbnail data
538                let mut buffer = Vec::new();
539                file.read_to_end(&mut buffer)?;
540
541                // Try to load as an image
542                if let Ok(img) = image::load_from_memory(&buffer) {
543                    let thumbnail = self.create_thumbnail(&img);
544                    let mut output = fs::File::create(output_path)?;
545                    let rgb = thumbnail.to_rgb8();
546                    let mut encoder = image::codecs::jpeg::JpegEncoder::new_with_quality(
547                        &mut output,
548                        THUMBNAIL_JPEG_QUALITY,
549                    );
550                    encoder
551                        .encode(
552                            &rgb,
553                            rgb.width(),
554                            rgb.height(),
555                            image::ExtendedColorType::Rgb8,
556                        )
557                        .map_err(|e| {
558                            HippoError::Other(format!("Failed to save Office thumbnail: {}", e))
559                        })?;
560                    return Ok(());
561                }
562            }
563        }
564
565        Err(HippoError::Other(
566            "No embedded thumbnail found in Office document".to_string(),
567        ))
568    }

668        let name1 = manager.get_thumbnail_name(Path::new("/path/to/image1.jpg"));
669        let name2 = manager.get_thumbnail_name(Path::new("/path/to/image2.jpg"));
670        let name3 = manager.get_thumbnail_name(Path::new("/path/to/image1.jpg"));
671
672        // Different paths should have different names
673        assert_ne!(name1, name2);
674        // Same path should have same name
675        assert_eq!(name1, name3);
676        // All should end with .jpg
677        assert!(name1.ends_with(".jpg"));
