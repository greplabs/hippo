11/// Size threshold for partial hashing (files larger than this use chunked hashing)
12const LARGE_FILE_THRESHOLD: u64 = 100 * 1024 * 1024; // 100MB
13
14/// Chunk size for reading files
15const CHUNK_SIZE: usize = 8 * 1024; // 8KB
16
17/// A group of duplicate files
18#[derive(Debug, Clone)]
19pub struct DuplicateGroup {
20    /// The shared content hash
21    pub hash: String,
22    /// File size (all duplicates have same size)
23    pub size: u64,
24    /// Memory IDs of duplicate files
25    pub memory_ids: Vec<MemoryId>,
26    /// File paths of duplicates
27    pub paths: Vec<std::path::PathBuf>,
28}
29
30impl DuplicateGroup {
31    /// Get the number of duplicates (excluding the original)
32    pub fn duplicate_count(&self) -> usize {
33        self.memory_ids.len().saturating_sub(1)
34    }
35
36    /// Get the total wasted space (duplicate_count * size)
37    pub fn wasted_bytes(&self) -> u64 {
38        self.duplicate_count() as u64 * self.size
39    }
40}

42/// Summary of duplicate detection results
43#[derive(Debug, Clone, Default)]
44pub struct DuplicateSummary {

46    pub files_scanned: usize,
47    /// Number of duplicate groups found
48    pub duplicate_groups: usize,
49    /// Total duplicate files (excluding originals)
50    pub total_duplicates: usize,
51    /// Total wasted space in bytes
52    pub wasted_bytes: u64,

55/// Compute SHA-256 hash of a file
56pub fn compute_file_hash(path: &Path) -> Result<String> {
57    let file = std::fs::File::open(path)
58        .map_err(|e| HippoError::Other(format!("Failed to open file: {}", e)))?;
59
60    let metadata = file
61        .metadata()
62        .map_err(|e| HippoError::Other(format!("Failed to get metadata: {}", e)))?;
63
64    let mut hasher = Sha256::new();
65    let mut reader = std::io::BufReader::new(file);
66
67    // For very large files, use a sampling strategy
68    if metadata.len() > LARGE_FILE_THRESHOLD {
69        // Hash first chunk, middle chunk, and last chunk
70        let mut buffer = vec![0u8; CHUNK_SIZE];
71
72        // First chunk
73        let n = reader
74            .read(&mut buffer)
75            .map_err(|e| HippoError::Other(format!("Read error: {}", e)))?;
76        hasher.update(&buffer[..n]);
77
78        // Include file size in hash for large files
79        hasher.update(metadata.len().to_le_bytes());
80    } else {
81        // Hash entire file for smaller files
82        let mut buffer = vec![0u8; CHUNK_SIZE];
83        loop {
84            let n = reader
85                .read(&mut buffer)
86                .map_err(|e| HippoError::Other(format!("Read error: {}", e)))?;
87            if n == 0 {
88                break;
89            }
90            hasher.update(&buffer[..n]);
91        }
92    }
93
94    let result = hasher.finalize();
95    Ok(format!("{:x}", result))
96}

98/// Compute a quick hash based on file size and first bytes (for pre-filtering)
99pub fn compute_quick_hash(path: &Path) -> Result<String> {
100    let file = std::fs::File::open(path)
101        .map_err(|e| HippoError::Other(format!("Failed to open file: {}", e)))?;
102
103    let metadata = file
104        .metadata()
105        .map_err(|e| HippoError::Other(format!("Failed to get metadata: {}", e)))?;
106
107    let mut hasher = Sha256::new();
108
109    // Include file size
110    hasher.update(metadata.len().to_le_bytes());
111
112    // Read first 4KB
113    let mut reader = std::io::BufReader::new(file);
114    let mut buffer = vec![0u8; 4096];
115    let n = reader
116        .read(&mut buffer)
117        .map_err(|e| HippoError::Other(format!("Read error: {}", e)))?;
118    hasher.update(&buffer[..n]);
119
120    let result = hasher.finalize();
121    Ok(format!("{:x}", result))
122}

124/// Find duplicates from a list of memories
125pub fn find_duplicates(
126    memories: &[Memory],
127    min_size: u64,
128) -> (Vec<DuplicateGroup>, DuplicateSummary) {
129    let mut summary = DuplicateSummary {
130        files_scanned: memories.len(),
131        ..Default::default()
132    };
133
134    // Group by hash
135    let mut hash_groups: HashMap<String, Vec<&Memory>> = HashMap::new();
136
137    for memory in memories {
138        // Skip small files
139        if memory.metadata.file_size < min_size {
140            continue;
141        }
142
143        // Use existing hash or compute one
144        if let Some(ref hash) = memory.metadata.hash {
145            hash_groups.entry(hash.clone()).or_default().push(memory);
146        }
147    }
148
149    // Find groups with duplicates
150    let mut duplicate_groups = Vec::new();
151
152    for (hash, group) in hash_groups {
153        if group.len() > 1 {
154            let size = group[0].metadata.file_size;
155            let dup_group = DuplicateGroup {
156                hash,
157                size,
158                memory_ids: group.iter().map(|m| m.id).collect(),
159                paths: group.iter().map(|m| m.path.clone()).collect(),
160            };
161
162            summary.total_duplicates += dup_group.duplicate_count();
163            summary.wasted_bytes += dup_group.wasted_bytes();
164            duplicate_groups.push(dup_group);
165        }
166    }
167
168    summary.duplicate_groups = duplicate_groups.len();
169
170    // Sort by wasted space (most wasted first)
171    duplicate_groups.sort_by_key(|g| std::cmp::Reverse(g.wasted_bytes()));
172
173    (duplicate_groups, summary)
174}

176/// Find duplicates by scanning file system and computing hashes
177pub fn find_duplicates_by_scanning(
178    memories: &[Memory],
179    min_size: u64,
180) -> Result<(Vec<DuplicateGroup>, DuplicateSummary)> {
181    let mut summary = DuplicateSummary {
182        files_scanned: 0,
183        ..Default::default()
184    };
185
186    // First pass: group by file size (quick filter)
187    let mut size_groups: HashMap<u64, Vec<&Memory>> = HashMap::new();
188
189    for memory in memories {
190        if memory.metadata.file_size >= min_size && memory.path.exists() {
191            size_groups
192                .entry(memory.metadata.file_size)
193                .or_default()
194                .push(memory);
195            summary.files_scanned += 1;
196        }
197    }
198
199    // Second pass: compute hashes only for size groups with potential duplicates
200    let mut hash_groups: HashMap<String, Vec<&Memory>> = HashMap::new();
201
202    for (_, group) in size_groups {
203        if group.len() < 2 {
204            continue; // No potential duplicates
205        }
206
207        for memory in group {
208            // Use existing hash or compute one
209            let hash = if let Some(ref h) = memory.metadata.hash {
210                h.clone()
211            } else {
212                match compute_file_hash(&memory.path) {
213                    Ok(h) => h,
214                    Err(_) => continue, // Skip files we can't hash
215                }
216            };
217
218            hash_groups.entry(hash).or_default().push(memory);
219        }
220    }
221
222    // Find groups with duplicates
223    let mut duplicate_groups = Vec::new();
224
225    for (hash, group) in hash_groups {
226        if group.len() > 1 {
227            let size = group[0].metadata.file_size;
228            let dup_group = DuplicateGroup {
229                hash,
230                size,
231                memory_ids: group.iter().map(|m| m.id).collect(),
232                paths: group.iter().map(|m| m.path.clone()).collect(),
233            };
234
235            summary.total_duplicates += dup_group.duplicate_count();
236            summary.wasted_bytes += dup_group.wasted_bytes();
237            duplicate_groups.push(dup_group);
238        }
239    }
240
241    summary.duplicate_groups = duplicate_groups.len();
242
243    // Sort by wasted space (most wasted first)
244    duplicate_groups.sort_by_key(|g| std::cmp::Reverse(g.wasted_bytes()));
245
246    Ok((duplicate_groups, summary))
247}

249/// Format bytes into human readable string
250pub fn format_bytes(bytes: u64) -> String {
251    const KB: u64 = 1024;
252    const MB: u64 = KB * 1024;
253    const GB: u64 = MB * 1024;
254
255    if bytes >= GB {
256        format!("{:.1} GB", bytes as f64 / GB as f64)
257    } else if bytes >= MB {
258        format!("{:.1} MB", bytes as f64 / MB as f64)
259    } else if bytes >= KB {
260        format!("{:.1} KB", bytes as f64 / KB as f64)
261    } else {
262        format!("{} B", bytes)
263    }
264}

266/// A group of semantically similar files (not exact duplicates)
267#[derive(Debug, Clone, serde::Serialize)]
268pub struct SimilarGroup {

270    pub representative_id: MemoryId,

298/// Find semantically similar files using embeddings
299///
300/// This uses vector embeddings to find files that are similar in content,
301/// even if they're not exact duplicates. For example:
302/// - Photos of the same subject from different angles
303/// - Documents covering the same topic
304/// - Code files with similar functionality
305pub fn find_similar_by_embedding(
306    memories: &[Memory],
307    embeddings: &HashMap<MemoryId, Vec<f32>>,
308    threshold: f32,
309    min_group_size: usize,
310) -> (Vec<SimilarGroup>, SimilarSummary) {
311    let mut summary = SimilarSummary {
312        files_analyzed: memories.len(),
313        threshold,
314        ..Default::default()
315    };
316
317    // Only consider memories that have embeddings
318    let memories_with_embeddings: Vec<_> = memories
319        .iter()
320        .filter(|m| embeddings.contains_key(&m.id))
321        .collect();
322
323    if memories_with_embeddings.is_empty() {
324        return (vec![], summary);
325    }
326
327    // Track which memories have been grouped
328    let mut grouped: std::collections::HashSet<MemoryId> = std::collections::HashSet::new();
329    let mut groups: Vec<SimilarGroup> = Vec::new();
330
331    // For each memory, find similar ones
332    for memory in &memories_with_embeddings {
333        if grouped.contains(&memory.id) {
334            continue;
335        }
336
337        let embedding = match embeddings.get(&memory.id) {
338            Some(e) => e,
339            None => continue,
340        };
341
342        let mut similar: Vec<(&Memory, f32)> = Vec::new();
343        similar.push((memory, 1.0)); // Include self
344
345        // Find all similar memories
346        for other in &memories_with_embeddings {
347            if other.id == memory.id || grouped.contains(&other.id) {
348                continue;
349            }
350
351            if let Some(other_embedding) = embeddings.get(&other.id) {
352                let similarity = cosine_similarity(embedding, other_embedding);
353                if similarity >= threshold {
354                    similar.push((other, similarity));
355                }
356            }
357        }
358
359        // Only create a group if we have enough similar files
360        if similar.len() >= min_group_size {
361            // Mark all as grouped
362            for (m, _) in &similar {
363                grouped.insert(m.id);
364            }
365
366            // Calculate average similarity
367            let avg_sim = similar.iter().map(|(_, s)| s).sum::<f32>() / similar.len() as f32;
368
369            // Determine file type
370            let file_type = match &memory.kind {
371                crate::MemoryKind::Image { .. } => "image",
372                crate::MemoryKind::Video { .. } => "video",
373                crate::MemoryKind::Audio { .. } => "audio",
374                crate::MemoryKind::Code { .. } => "code",
375                crate::MemoryKind::Document { .. } => "document",
376                _ => "file",
377            };
378
379            let group = SimilarGroup {
380                representative_id: memory.id,
381                representative_name: memory
382                    .path
383                    .file_name()
384                    .map(|n| n.to_string_lossy().to_string())
385                    .unwrap_or_default(),
386                memory_ids: similar.iter().map(|(m, _)| m.id).collect(),
387                paths: similar.iter().map(|(m, _)| m.path.clone()).collect(),
388                similarity_scores: similar.iter().map(|(_, s)| *s).collect(),
389                avg_similarity: avg_sim,
390                file_type: file_type.to_string(),
391            };
392
393            summary.total_similar += group.memory_ids.len();
394            groups.push(group);
395        }
396    }
397
398    summary.similar_groups = groups.len();
399
400    // Sort by group size (largest first)
401    groups.sort_by_key(|g| std::cmp::Reverse(g.memory_ids.len()));
402
403    (groups, summary)
404}

406/// Compute cosine similarity between two vectors
407fn cosine_similarity(a: &[f32], b: &[f32]) -> f32 {
408    if a.len() != b.len() || a.is_empty() {
409        return 0.0;
410    }
411
412    let dot: f32 = a.iter().zip(b.iter()).map(|(x, y)| x * y).sum();
413    let norm_a: f32 = a.iter().map(|x| x * x).sum::<f32>().sqrt();
414    let norm_b: f32 = b.iter().map(|x| x * x).sum::<f32>().sqrt();
415
416    if norm_a == 0.0 || norm_b == 0.0 {
417        return 0.0;
418    }
419
420    dot / (norm_a * norm_b)
421}
