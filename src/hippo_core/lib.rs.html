1//! # Hippo Core
2//!
3//! The memory that never forgets. 🦛
4//!
5//! Hippo is an intelligent file memory system that indexes, understands,
6//! and connects all your files across local storage and cloud providers.
7
8#![allow(missing_docs)]
9//!
10//! ## Architecture
11//!
12//! ```text
13//! ┌─────────────────────────────────────────────────────────────┐
14//! │                         Hippo Core                          │
15//! ├─────────────────────────────────────────────────────────────┤
16//! │                                                             │
17//! │  ┌─────────┐  ┌─────────┐  ┌─────────┐  ┌─────────────────┐│
18//! │  │ Sources │  │ Indexer │  │Embedder │  │   Search        ││
19//! │  │         │──│         │──│  (CLIP) │──│   (Qdrant)      ││
20//! │  │ Local   │  │ Extract │  │  (BGE)  │  │                 ││
21//! │  │ GDrive  │  │ Parse   │  │  (Code) │  │ Semantic        ││
22//! │  │ iCloud  │  │ Analyze │  │         │  │ Tag-based       ││
23//! │  │ S3      │  │         │  │         │  │ Filters         ││
24//! │  └─────────┘  └─────────┘  └─────────┘  └─────────────────┘│
25//! │       │            │            │               │          │
26//! │       ▼            ▼            ▼               ▼          │
27//! │  ┌──────────────────────────────────────────────────────┐  │
28//! │  │                    Storage Layer                      │  │
29//! │  │  SQLite (metadata)  +  Qdrant (vectors)              │  │
30//! │  └──────────────────────────────────────────────────────┘  │
31//! │                            │                               │
32//! │                            ▼                               │
33//! │  ┌──────────────────────────────────────────────────────┐  │
34//! │  │                   Knowledge Graph                     │  │
35//! │  │  Connections • Clusters • Mind Maps                  │  │
36//! │  └──────────────────────────────────────────────────────┘  │
37//! │                                                             │
38//! └─────────────────────────────────────────────────────────────┘
39//! ```
40//!
41//! ## Quick Start
42//!
43//! ```rust,ignore
44//! use hippo_core::Hippo;
45//!
46//! #[tokio::main]
47//! async fn main() -> anyhow::Result<()> {
48//!     // Initialize Hippo
49//!     let hippo = Hippo::new().await?;
50//!     
51//!     // Add a local folder to index
52//!     hippo.add_source(Source::Local {
53//!         root_path: "/Users/you/Photos".into()
54//!     }).await?;
55//!     
56//!     // Search semantically
57//!     let results = hippo.search("sunset beach vacation").await?;
58//!     
59//!     // Add tags inline
60//!     hippo.add_tag(results[0].id, Tag::user("favorites")).await?;
61//!     
62//!     Ok(())
63//! }
64//! ```

175impl Default for HippoConfig {
176    fn default() -> Self {
177        let data_dir = directories::ProjectDirs::from("", "", "Hippo")
178            .map(|d| d.data_dir().to_path_buf())
179            .unwrap_or_else(|| PathBuf::from(".hippo"));
180
181        // Allow up to 16 cores for I/O-bound file indexing operations
182        // Most operations are I/O bound, so higher parallelism helps
183        let parallelism = num_cpus::get().min(16);
184
185        Self {
186            data_dir,
187            local_embeddings: true,
188            ai_api_key: None,
189            qdrant_url: "http://localhost:6334".into(),
190            indexing_parallelism: parallelism,
191            auto_tag_enabled: true,
192        }
193    }
194}

202    /// Create a new Hippo instance with custom configuration
203    pub async fn with_config(config: HippoConfig) -> Result<Self> {
204        // Ensure data directory exists
205        std::fs::create_dir_all(&config.data_dir)?;
206
207        // Initialize components
208        let storage = Arc::new(storage::Storage::new(&config).await?);
209        let embedder = Arc::new(embeddings::Embedder::new(&config).await?);
210        let indexer = Arc::new(indexer::Indexer::new(
211            storage.clone(),
212            embedder.clone(),
213            &config,
214        )?);
215        let searcher =
216            Arc::new(search::Searcher::new(storage.clone(), embedder.clone(), &config).await?);
217        let graph = Arc::new(RwLock::new(graph::KnowledgeGraph::new(storage.clone())));
218
219        // Initialize watcher (optional - can be started later)
220        let watcher = match watcher::FileWatcher::new(storage.clone(), None) {
221            Ok(mut w) => {
222                // Set the indexer so the watcher can re-index changed files
223                w.set_indexer(indexer.clone());
224                Some(Arc::new(RwLock::new(w)))
225            }
226            Err(e) => {
227                tracing::warn!("Failed to create file watcher: {}", e);
228                None
229            }
230        };
231
232        // Initialize thumbnail manager
233        let thumbnail_manager = Arc::new(thumbnails::ThumbnailManager::new()?);
234
235        // Initialize organizer
236        let organizer = Arc::new(organization::Organizer::new(
237            storage.clone(),
238            embedder.clone(),
239        ));
240
241        Ok(Self {
242            storage,
243            indexer,
244            embedder,
245            searcher,
246            graph,
247            watcher,
248            thumbnail_manager,
249            organizer,
250            config,
251        })
252    }

266            // Delete all memories from this source's path

271            }
272        }
273        self.storage.remove_source(source).await?;
274        Ok(())
275    }

399    /// Start watching a source for file changes
400    pub async fn watch_source(&self, source: &Source) -> Result<()> {
401        if let Some(watcher) = &self.watcher {
402            if let Source::Local { root_path } = source {
403                watcher.write().await.watch(root_path, source.clone()).await
404            } else {
405                Err(HippoError::Other(
406                    "Only local sources can be watched".to_string(),
407                ))
408            }
409        } else {
410            Err(HippoError::Other("File watcher not available".to_string()))
411        }
412    }
413
414    /// Stop watching a source
415    pub async fn unwatch_source(&self, source: &Source) -> Result<()> {
416        if let Some(watcher) = &self.watcher {
417            if let Source::Local { root_path } = source {
418                watcher.write().await.unwatch(root_path).await
419            } else {
420                Ok(())
421            }
422        } else {
423            Ok(())
424        }
425    }

479    /// Subscribe to file watch events
480    pub fn subscribe_watch_events(
481        &self,
482    ) -> Option<tokio::sync::broadcast::Receiver<watcher::WatchEvent>> {
483        self.watcher.as_ref().map(|w| {
484            // This is a bit tricky - we need to get the receiver without blocking
485            // For now, return None - we'll handle events differently in Tauri
486            // The proper way would be to use a channel that can be subscribed to multiple times
487            // We'll emit events via Tauri's event system instead
488            tokio::task::block_in_place(|| {
489                tokio::runtime::Handle::current().block_on(async { w.read().await.subscribe() })
490            })
491        })
492    }

686    /// Full storage optimization: cleanup orphans, vacuum, and reclaim space
687    pub async fn optimize_storage(&self) -> Result<VacuumStats> {
688        // First clean up orphaned embeddings
689        let orphans = self.cleanup_orphaned_embeddings().await?;
690        tracing::info!("Cleaned up {} orphaned embeddings", orphans);
691
692        // Then vacuum to reclaim space
693        self.vacuum().await
694    }
695}

700        std::thread::available_parallelism()
701            .map(|p| p.get())
702            .unwrap_or(4)
703    }
