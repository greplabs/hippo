12/// Extract metadata from a file based on its type
13pub fn extract_metadata(path: &Path, kind: &MemoryKind) -> Result<MemoryMetadata> {
14    let mut metadata = MemoryMetadata {
15        title: path.file_stem().and_then(|s| s.to_str()).map(String::from),
16        ..Default::default()
17    };
18
19    match kind {
20        MemoryKind::Image { .. } => {
21            extract_image_metadata(path, &mut metadata)?;
22        }
23        MemoryKind::Video { .. } => {
24            extract_video_metadata(path, &mut metadata)?;
25        }
26        MemoryKind::Document { format, .. } => {
27            extract_document_metadata(path, format, &mut metadata)?;
28        }
29        MemoryKind::Code { language, .. } => {
30            extract_code_metadata(path, language, &mut metadata)?;
31        }
32        MemoryKind::Audio { .. } => {
33            extract_audio_metadata(path, &mut metadata)?;
34        }
35        _ => {}
36    }
37
38    Ok(metadata)
39}

41/// Extract EXIF and other metadata from images
42fn extract_image_metadata(path: &Path, metadata: &mut MemoryMetadata) -> Result<()> {
43    let file = File::open(path)?;
44    let mut reader = BufReader::new(file);
45
46    // Try to read EXIF data
47    if let Ok(exif_reader) = exif::Reader::new().read_from_container(&mut reader) {
48        let mut exif_data = ExifData {
49            camera_make: None,
50            camera_model: None,
51            lens: None,
52            focal_length: None,
53            aperture: None,
54            iso: None,
55            shutter_speed: None,
56            exposure_time: None,
57            taken_at: None,
58            gps: None,
59        };
60
61        // Camera info
62        if let Some(field) = exif_reader.get_field(exif::Tag::Make, exif::In::PRIMARY) {
63            exif_data.camera_make = Some(field.display_value().to_string());
64        }
65        if let Some(field) = exif_reader.get_field(exif::Tag::Model, exif::In::PRIMARY) {
66            exif_data.camera_model = Some(field.display_value().to_string());
67        }
68
69        // Lens
70        if let Some(field) = exif_reader.get_field(exif::Tag::LensModel, exif::In::PRIMARY) {
71            exif_data.lens = Some(field.display_value().to_string());
72        }
73
74        // Exposure settings
75        if let Some(field) = exif_reader.get_field(exif::Tag::FocalLength, exif::In::PRIMARY) {
76            if let exif::Value::Rational(ref v) = field.value {
77                if let Some(r) = v.first() {
78                    exif_data.focal_length = Some(r.num as f32 / r.denom as f32);
79                }
80            }
81        }
82
83        if let Some(field) = exif_reader.get_field(exif::Tag::FNumber, exif::In::PRIMARY) {
84            if let exif::Value::Rational(ref v) = field.value {
85                if let Some(r) = v.first() {
86                    exif_data.aperture = Some(r.num as f32 / r.denom as f32);
87                }
88            }
89        }
90
91        if let Some(field) =
92            exif_reader.get_field(exif::Tag::PhotographicSensitivity, exif::In::PRIMARY)
93        {
94            if let exif::Value::Short(ref v) = field.value {
95                exif_data.iso = v.first().map(|&v| v as u32);
96            }
97        }
98
99        if let Some(field) = exif_reader.get_field(exif::Tag::ExposureTime, exif::In::PRIMARY) {
100            exif_data.shutter_speed = Some(field.display_value().to_string());
101            // Also store numeric value in seconds
102            if let exif::Value::Rational(ref v) = field.value {
103                if let Some(r) = v.first() {
104                    exif_data.exposure_time = Some(r.num as f32 / r.denom as f32);
105                }
106            }
107        }
108
109        // Date taken
110        if let Some(field) = exif_reader.get_field(exif::Tag::DateTimeOriginal, exif::In::PRIMARY) {
111            let date_str = field.display_value().to_string();
112            // Parse "2024:07:15 14:30:00" format
113            if let Ok(dt) = chrono::NaiveDateTime::parse_from_str(
114                date_str.trim_matches('"'),
115                "%Y:%m:%d %H:%M:%S",
116            ) {
117                exif_data.taken_at =
118                    Some(chrono::DateTime::from_naive_utc_and_offset(dt, chrono::Utc));
119            }
120        }
121
122        // GPS coordinates
123        let lat = extract_gps_coord(
124            &exif_reader,
125            exif::Tag::GPSLatitude,
126            exif::Tag::GPSLatitudeRef,
127        );
128        let lon = extract_gps_coord(
129            &exif_reader,
130            exif::Tag::GPSLongitude,
131            exif::Tag::GPSLongitudeRef,
132        );
133
134        if let (Some(lat), Some(lon)) = (lat, lon) {
135            // Extract altitude if available
136            let altitude = extract_gps_altitude(&exif_reader);
137
138            exif_data.gps = Some(GeoLocation {
139                latitude: lat,
140                longitude: lon,
141                altitude,
142                place_name: None,
143                city: None,
144                country: None,
145            });
146            metadata.location = exif_data.gps.clone();
147        }
148
149        metadata.exif = Some(exif_data);
150    }
151
152    Ok(())
153}

155/// Extract GPS coordinate from EXIF
156fn extract_gps_coord(exif: &exif::Exif, coord_tag: exif::Tag, ref_tag: exif::Tag) -> Option<f64> {
157    let coord_field = exif.get_field(coord_tag, exif::In::PRIMARY)?;
158    let ref_field = exif.get_field(ref_tag, exif::In::PRIMARY)?;
159
160    if let exif::Value::Rational(ref rationals) = coord_field.value {
161        if rationals.len() >= 3 {
162            let degrees = rationals[0].num as f64 / rationals[0].denom as f64;
163            let minutes = rationals[1].num as f64 / rationals[1].denom as f64;
164            let seconds = rationals[2].num as f64 / rationals[2].denom as f64;
165
166            let mut coord = degrees + minutes / 60.0 + seconds / 3600.0;
167
168            // Check reference (N/S or E/W)
169            let ref_str = ref_field.display_value().to_string();
170            if ref_str.contains('S') || ref_str.contains('W') {
171                coord = -coord;
172            }
173
174            return Some(coord);
175        }
176    }
177
178    None
179}

181/// Extract GPS altitude from EXIF
182fn extract_gps_altitude(exif: &exif::Exif) -> Option<f64> {
183    let altitude_field = exif.get_field(exif::Tag::GPSAltitude, exif::In::PRIMARY)?;
184
185    if let exif::Value::Rational(ref rationals) = altitude_field.value {
186        if let Some(r) = rationals.first() {
187            let mut altitude = r.num as f64 / r.denom as f64;
188
189            // Check altitude reference (0 = above sea level, 1 = below sea level)
190            if let Some(ref_field) = exif.get_field(exif::Tag::GPSAltitudeRef, exif::In::PRIMARY) {
191                if let exif::Value::Byte(ref bytes) = ref_field.value {
192                    if let Some(&1) = bytes.first() {
193                        altitude = -altitude;
194                    }
195                }
196            }
197
198            return Some(altitude);
199        }
200    }
201
202    None
203}

205/// Extract metadata from video files using ffprobe
206fn extract_video_metadata(path: &Path, metadata: &mut MemoryMetadata) -> Result<()> {
207    use std::process::Command;
208
209    let output = Command::new("ffprobe")
210        .args([
211            "-v",
212            "error",
213            "-show_entries",
214            "stream=codec_name,codec_type,bit_rate,width,height,r_frame_rate,sample_rate,channels:format=bit_rate",
215            "-of",
216            "json",
217            path.to_str().unwrap_or(""),
218        ])
219        .output();
220
221    if let Ok(output) = output {
222        if output.status.success() {
223            let json_str = String::from_utf8_lossy(&output.stdout);
224            if let Ok(json) = serde_json::from_str::<serde_json::Value>(&json_str) {
225                let mut video_meta = VideoMetadata {
226                    codec: None,
227                    bitrate: None,
228                    framerate: None,
229                    width: None,
230                    height: None,
231                    audio_codec: None,
232                    audio_channels: None,
233                    audio_sample_rate: None,
234                };
235
236                // Extract format-level bitrate
237                if let Some(format) = json.get("format") {
238                    if let Some(bitrate_str) = format.get("bit_rate").and_then(|v| v.as_str()) {
239                        if let Ok(bitrate) = bitrate_str.parse::<u64>() {
240                            video_meta.bitrate = Some(bitrate);
241                        }
242                    }
243                }
244
245                // Extract stream information
246                if let Some(streams) = json.get("streams").and_then(|v| v.as_array()) {
247                    for stream in streams {
248                        let codec_type = stream
249                            .get("codec_type")
250                            .and_then(|v| v.as_str())
251                            .unwrap_or("");
252
253                        match codec_type {
254                            "video" => {
255                                // Video codec
256                                if let Some(codec) =
257                                    stream.get("codec_name").and_then(|v| v.as_str())
258                                {
259                                    video_meta.codec = Some(codec.to_string());
260                                }
261
262                                // Dimensions
263                                if let Some(width) = stream.get("width").and_then(|v| v.as_u64()) {
264                                    video_meta.width = Some(width as u32);
265                                }
266                                if let Some(height) = stream.get("height").and_then(|v| v.as_u64())
267                                {
268                                    video_meta.height = Some(height as u32);
269                                }
270
271                                // Frame rate (parse "30/1" or "24000/1001" format)
272                                if let Some(fps_str) =
273                                    stream.get("r_frame_rate").and_then(|v| v.as_str())
274                                {
275                                    if let Some((num, den)) = fps_str.split_once('/') {
276                                        if let (Ok(n), Ok(d)) =
277                                            (num.parse::<f32>(), den.parse::<f32>())
278                                        {
279                                            if d > 0.0 {
280                                                video_meta.framerate = Some(n / d);
281                                            }
282                                        }
283                                    }
284                                }
285
286                                // Stream-level bitrate (if format-level not available)
287                                if video_meta.bitrate.is_none() {
288                                    if let Some(bitrate_str) =
289                                        stream.get("bit_rate").and_then(|v| v.as_str())
290                                    {
291                                        if let Ok(bitrate) = bitrate_str.parse::<u64>() {
292                                            video_meta.bitrate = Some(bitrate);
293                                        }
294                                    }
295                                }
296                            }
297                            "audio" => {
298                                // Audio codec
299                                if let Some(codec) =
300                                    stream.get("codec_name").and_then(|v| v.as_str())
301                                {
302                                    video_meta.audio_codec = Some(codec.to_string());
303                                }
304
305                                // Sample rate
306                                if let Some(sample_rate_str) =
307                                    stream.get("sample_rate").and_then(|v| v.as_str())
308                                {
309                                    if let Ok(sample_rate) = sample_rate_str.parse::<u32>() {
310                                        video_meta.audio_sample_rate = Some(sample_rate);
311                                    }
312                                }
313
314                                // Channels
315                                if let Some(channels) =
316                                    stream.get("channels").and_then(|v| v.as_u64())
317                                {
318                                    video_meta.audio_channels = Some(channels as u32);
319                                }
320                            }
321                            _ => {}
322                        }
323                    }
324                }
325
326                metadata.video_metadata = Some(video_meta);
327            }
328        }
329    }
330
331    Ok(())
332}

334/// Extract text and metadata from documents
335fn extract_document_metadata(
336    path: &Path,
337    format: &DocumentFormat,
338    metadata: &mut MemoryMetadata,
339) -> Result<()> {
340    match format {
341        DocumentFormat::PlainText | DocumentFormat::Markdown => {
342            let content = std::fs::read_to_string(path)?;
343            metadata.word_count = Some(content.split_whitespace().count() as u32);
344            metadata.text_preview = Some(content.chars().take(MAX_TEXT_PREVIEW_CHARS).collect());
345        }
346        DocumentFormat::Pdf => {
347            // PDF extraction would use pdf-extract crate
348            // For now, just note it's a PDF
349        }
350        _ => {}
351    }
352
353    Ok(())
354}

356/// Extract metadata from code files
357fn extract_code_metadata(path: &Path, language: &str, metadata: &mut MemoryMetadata) -> Result<()> {
358    let content = std::fs::read_to_string(path)?;
359
360    let mut code_info = CodeInfo {
361        language: language.to_string(),
362        lines_of_code: content.lines().count() as u32,
363        imports: Vec::new(),
364        exports: Vec::new(),
365        functions: Vec::new(),
366        dependencies: Vec::new(),
367    };
368
369    // Simple import extraction (language-specific parsing would be better)
370    for line in content.lines() {
371        let trimmed = line.trim();
372
373        // Rust
374        if trimmed.starts_with("use ") {
375            if let Some(import) = trimmed
376                .strip_prefix("use ")
377                .and_then(|s| s.strip_suffix(';'))
378            {
379                code_info.imports.push(import.to_string());
380            }
381        }
382        // Python, JavaScript/TypeScript imports
383        else if trimmed.starts_with("import ")
384            || trimmed.starts_with("from ")
385            || trimmed.starts_with("require(")
386        {
387            code_info.imports.push(trimmed.to_string());
388        }
389
390        // Simple function detection
391        if trimmed.starts_with("fn ") || trimmed.starts_with("pub fn ") {
392            if let Some(name) = extract_function_name(trimmed, "fn ") {
393                code_info.functions.push(FunctionInfo {
394                    name,
395                    line_start: 0, // Would need proper parsing
396                    line_end: 0,
397                    is_public: trimmed.starts_with("pub "),
398                    doc_comment: None,
399                });
400            }
401        } else if trimmed.starts_with("def ") {
402            if let Some(name) = extract_function_name(trimmed, "def ") {
403                code_info.functions.push(FunctionInfo {
404                    name,
405                    line_start: 0,
406                    line_end: 0,
407                    is_public: true,
408                    doc_comment: None,
409                });
410            }
411        } else if trimmed.starts_with("function ") || trimmed.contains("=> {") {
412            if let Some(name) = extract_function_name(trimmed, "function ") {
413                code_info.functions.push(FunctionInfo {
414                    name,
415                    line_start: 0,
416                    line_end: 0,
417                    is_public: true,
418                    doc_comment: None,
419                });
420            }
421        }
422    }
423
424    metadata.code_info = Some(code_info);
425    metadata.text_preview = Some(content.chars().take(MAX_TEXT_PREVIEW_CHARS).collect());
426
427    Ok(())
428}

430fn extract_function_name(line: &str, prefix: &str) -> Option<String> {
431    let after_prefix = line.split(prefix).nth(1)?;
432    let name: String = after_prefix
433        .chars()
434        .take_while(|c| c.is_alphanumeric() || *c == '_')
435        .collect();
436
437    if name.is_empty() {
438        None
439    } else {
440        Some(name)
441    }
442}

444/// Extract metadata from audio files using symphonia
445fn extract_audio_metadata(path: &Path, metadata: &mut MemoryMetadata) -> Result<()> {
446    use symphonia::core::codecs::CODEC_TYPE_NULL;
447    use symphonia::core::formats::FormatOptions;
448    use symphonia::core::io::MediaSourceStream;
449    use symphonia::core::meta::MetadataOptions;
450    use symphonia::core::probe::Hint;
451
452    // Try to open the file
453    let file = std::fs::File::open(path)?;
454    let mss = MediaSourceStream::new(Box::new(file), Default::default());
455
456    // Create a hint based on file extension
457    let mut hint = Hint::new();
458    if let Some(ext) = path.extension().and_then(|e| e.to_str()) {
459        hint.with_extension(ext);
460    }
461
462    // Probe the media source
463    let format_opts = FormatOptions::default();
464    let metadata_opts = MetadataOptions::default();
465
466    if let Ok(mut probed) =
467        symphonia::default::get_probe().format(&hint, mss, &format_opts, &metadata_opts)
468    {
469        let mut audio_meta = AudioMetadata {
470            codec: None,
471            bitrate: None,
472            sample_rate: None,
473            channels: None,
474            artist: None,
475            album: None,
476            title: None,
477            track_number: None,
478            genre: None,
479            year: None,
480        };
481
482        // Get the default track (usually the first audio track)
483        if let Some(track) = probed
484            .format
485            .tracks()
486            .iter()
487            .find(|t| t.codec_params.codec != CODEC_TYPE_NULL)
488        {
489            // Codec
490            let codec_registry = symphonia::default::get_codecs();
491            if let Some(codec_desc) = codec_registry.get_codec(track.codec_params.codec) {
492                audio_meta.codec = Some(codec_desc.short_name.to_string());
493            }
494
495            // Sample rate
496            if let Some(sample_rate) = track.codec_params.sample_rate {
497                audio_meta.sample_rate = Some(sample_rate);
498            }
499
500            // Channels
501            if let Some(channels) = track.codec_params.channels {
502                audio_meta.channels = Some(channels.count() as u32);
503            }
504
505            // Bitrate (if available)
506            if let Some(n_frames) = track.codec_params.n_frames {
507                if let Some(sample_rate) = track.codec_params.sample_rate {
508                    if let Some(_bits_per_sample) = track.codec_params.bits_per_sample {
509                        if let Some(_channels) = track.codec_params.channels {
510                            // Calculate approximate bitrate
511                            let duration_secs = n_frames as f64 / sample_rate as f64;
512                            if duration_secs > 0.0 {
513                                let file_size =
514                                    std::fs::metadata(path).map(|m| m.len()).unwrap_or(0);
515                                audio_meta.bitrate = Some((file_size * 8) / duration_secs as u64);
516                            }
517                        }
518                    }
519                }
520            }
521        }
522
523        // Extract metadata tags (ID3, Vorbis comments, etc.)
524        let metadata_rev = probed.format.metadata();
525
526        // Try getting metadata from the format's metadata
527        if let Some(metadata) = metadata_rev.current() {
528            for tag in metadata.tags() {
529                match tag.std_key {
530                    Some(symphonia::core::meta::StandardTagKey::Artist) => {
531                        audio_meta.artist = Some(tag.value.to_string());
532                    }
533                    Some(symphonia::core::meta::StandardTagKey::Album) => {
534                        audio_meta.album = Some(tag.value.to_string());
535                    }
536                    Some(symphonia::core::meta::StandardTagKey::TrackTitle) => {
537                        audio_meta.title = Some(tag.value.to_string());
538                    }
539                    Some(symphonia::core::meta::StandardTagKey::TrackNumber) => {
540                        if let Ok(num) = tag.value.to_string().parse::<u32>() {
541                            audio_meta.track_number = Some(num);
542                        }
543                    }
544                    Some(symphonia::core::meta::StandardTagKey::Genre) => {
545                        audio_meta.genre = Some(tag.value.to_string());
546                    }
547                    Some(symphonia::core::meta::StandardTagKey::Date) => {
548                        // Try to extract year from date
549                        let date_str = tag.value.to_string();
550                        if let Some(year_str) = date_str.split('-').next() {
551                            if let Ok(year) = year_str.parse::<u32>() {
552                                audio_meta.year = Some(year);
553                            }
554                        }
555                    }
556                    _ => {}
557                }
558            }
559        }
560
561        // Also check probed metadata (some formats store metadata differently)
562        if let Some(metadata) = probed.metadata.get() {
563            if let Some(current) = metadata.current() {
564                for tag in current.tags() {
565                    // Fill in any missing fields
566                    match tag.std_key {
567                        Some(symphonia::core::meta::StandardTagKey::Artist)
568                            if audio_meta.artist.is_none() =>
569                        {
570                            audio_meta.artist = Some(tag.value.to_string());
571                        }
572                        Some(symphonia::core::meta::StandardTagKey::Album)
573                            if audio_meta.album.is_none() =>
574                        {
575                            audio_meta.album = Some(tag.value.to_string());
576                        }
577                        Some(symphonia::core::meta::StandardTagKey::TrackTitle)
578                            if audio_meta.title.is_none() =>
579                        {
580                            audio_meta.title = Some(tag.value.to_string());
581                        }
582                        _ => {}
583                    }
584                }
585            }
586        }
587
588        // Set title in metadata if found
589        if let Some(title) = &audio_meta.title {
590            metadata.title = Some(title.clone());
591        }
592
593        metadata.audio_metadata = Some(audio_meta);
594    }
595
596    Ok(())
597}
