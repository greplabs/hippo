221fn format_bytes(bytes: u64) -> String {
222    const KB: u64 = 1024;
223    const MB: u64 = KB * 1024;
224    const GB: u64 = MB * 1024;
225
226    if bytes >= GB {
227        format!("{:.1} GB", bytes as f64 / GB as f64)
228    } else if bytes >= MB {
229        format!("{:.1} MB", bytes as f64 / MB as f64)
230    } else if bytes >= KB {
231        format!("{:.1} KB", bytes as f64 / KB as f64)
232    } else {
233        format!("{} B", bytes)
234    }
235}

271fn format_tags(tags: &[hippo_core::Tag]) -> String {
272    if tags.is_empty() {
273        return "‚Äî".dimmed().to_string();
274    }
275    tags.iter()
276        .map(|t| match t.source {
277            hippo_core::TagSource::User => t.name.bright_cyan().to_string(),
278            hippo_core::TagSource::Ai => format!("{}{}", t.name.bright_magenta(), "‚ú®".dimmed()),
279            hippo_core::TagSource::Imported => t.name.bright_green().to_string(),
280            hippo_core::TagSource::System => t.name.dimmed().to_string(),
281        })
282        .collect::<Vec<_>>()
283        .join(", ")
284}

319fn format_bytes_colored(bytes: u64) -> String {
320    const KB: u64 = 1024;
321    const MB: u64 = KB * 1024;
322    const GB: u64 = MB * 1024;
323
324    let formatted = format_bytes(bytes);
325
326    // Color based on size: small=green, medium=yellow, large=red
327    if bytes >= GB {
328        formatted.bright_red().to_string()
329    } else if bytes >= 100 * MB {
330        formatted.bright_yellow().to_string()
331    } else if bytes >= 10 * MB {
332        formatted.yellow().to_string()
333    } else {
334        formatted.bright_green().to_string()
335    }
336}

338fn print_memory_card(mem: &hippo_core::Memory, index: Option<usize>) {
339    let name = mem.metadata.title.clone().unwrap_or_else(|| {
340        mem.path
341            .file_name()
342            .map(|n| n.to_string_lossy().to_string())
343            .unwrap_or_else(|| "Unknown".to_string())
344    });
345
346    let prefix = match index {
347        Some(i) => format!("{}", format!("[{}]", i + 1).dimmed()),
348        None => "  ".to_string(),
349    };
350
351    // First line: icon + name + size
352    println!(
353        "{} {} {} {}",
354        prefix,
355        get_kind_icon(&mem.kind),
356        name.bright_white().bold(),
357        format!("({})", format_bytes_colored(mem.metadata.file_size)).dimmed()
358    );
359
360    // Second line: path (truncated if too long)
361    let path_str = mem.path.display().to_string();
362    let display_path = if path_str.len() > 60 {
363        format!("...{}", &path_str[path_str.len() - 57..])
364    } else {
365        path_str
366    };
367    println!("{}   {}", prefix, display_path.dimmed());
368
369    // Third line: tags (if any)
370    if !mem.tags.is_empty() {
371        println!("{}   {}", prefix, format_tags(&mem.tags));
372    }
373    println!();
374}

417        Commands::Sniff {
418            query,

426                "üß† semantic"

428                "üîç text"
429            };
430            print_header(&format!(
431                "Sniffing ({}) for \"{}\"",
432                search_mode,
433                query.bright_yellow()
434            ));
435
436            let results = if semantic {
437                // Use semantic search
438                hippo.semantic_search(&query, limit).await?
439            } else {
440                // Use text search with sort
441                let sort_order = match sort.to_lowercase().as_str() {
442                    "oldest" => hippo_core::SortOrder::DateOldest,
443                    "name" | "name-asc" => hippo_core::SortOrder::NameAsc,
444                    "name-desc" => hippo_core::SortOrder::NameDesc,
445                    "size" | "size-desc" => hippo_core::SortOrder::SizeDesc,
446                    "size-asc" => hippo_core::SortOrder::SizeAsc,

451                    text: Some(query.clone()),

453                        .into_iter()
454                        .map(|t| TagFilter {
455                            tag: t,
456                            mode: TagFilterMode::Include,

459                    limit,
460                    sort: sort_order,
461                    ..Default::default()
462                };
463                hippo.search_advanced(search_query).await?
